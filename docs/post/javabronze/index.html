インタフェース

// インタフェースは情報隠蔽を実現方法として使用します(公開する部分を担当)
// 情報隠蔽は、抽象化を維持する目的（非公開と公開明確に分け）
// インタフェースはインスタンス化できない！
//　インタフェースに定義できるのは、定数フィールドのみ
static final value;//定数フィールド
// インタフェースの定義と実現
public interface Sample{
   void sample();
}
// 実現 implements Sample
public Class SampleImpl implements Sample{
	public void sample(){
		 // code
	}
}
// 複数 implements Sample,Test
public Class SampleImpl implements Sample,Test{
	public void sample(){
		// code
	}
  public void test(){
		 // code
  }
}
/*
クラスは複数のインタフェースを実現できる
インタフェースは複数のインタフェースを継承できる
クラスは複数のクラスを継承できない　
*/

抽象メソッド

// インタフェースに定義した抽象メソッドは、暗黙的にpublic abstract　で修飾され
public interface Sample{
	 void test(String t);// 抽象メソッド
}
// 抽象クラス 
public abstract class Test{
	private String t_name;
	public Test(String t_name){
	  this.t_name  = t_name;	  
  }
  public String getName(){
			return t_name;
  }
  public void check(){
     System.out.println(this.t_name);  
  }
}
/*
抽象クラスの特徴
1.抽象クラスは、クラスとインタフェース両方の性質を持つクラス
2.抽象クラスは インスタンスできない ！
3.定義したメソッドは、抽象クラスを継承する具象クラスが実装する
4.抽象クラスに定義するメソッドは、abstractで修飾しない と コンパイルエラー！
*/

「６：２０」と表示される

public static void main(String[] args) {
        int a = 12;
        int b  = 8 ;
        if(a>=10||b>=10){　// a >=10 --> true
            a /= 2; // a = 6
            b+=a;  // b = 6+8 = 14
        }else // <---{ なし
            a /=2; // } によって

            b +=a; // 14+6 = 20
            System.out.println(a+":"+b); // 6 : 20
    }

10:10と表示される

public static void main(String[] args) {
            int i = 5;
							// i = 5 + 5  -->i+=5
            // iを使用した後で +1; i++
            // iを+1した後で使用する; ++i
            System.out.println((i+=5)+":"+(i--));
            // 10:10
            System.out.println((i+=5)+":"+(--i));
            // 10:9
    }
